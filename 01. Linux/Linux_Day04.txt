#!/bin/bash

# Update packages
dnf update -y

# Install Apache (httpd) and NGINX
dnf install -y httpd nginx

# Enable services at boot
systemctl enable httpd
systemctl enable nginx

# Start services
systemctl start httpd
systemctl start nginx

# Optional: create test pages
echo "<h1>Apache HTTPD Running</h1>" > /var/www/html/index.html
echo "<h1>NGINX Running</h1>" > /usr/share/nginx/html/index.html
